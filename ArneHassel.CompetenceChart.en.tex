  \addtocounter{wlm}{1}

  #1
  \forloop{ct}{1}{\value{ct} < \value{klm}}{\value{ct} = \value{kl}}%
  {%
    & \cellcolor[gray]{0.5}
  }
  \forloop{ct}{\value{klm}}{\value{ct} < \value{wlm}}{\value{ct} = \value{wl}}%
  {%
    & \cellcolor[gray]{0.9}
  }
  \forloop{ct}{\value{wlm}}{\value{ct} < 11}{\value{ct} = 11}%
  {%
    & \cellcolor[gray]{1}
  }
  \\
}
